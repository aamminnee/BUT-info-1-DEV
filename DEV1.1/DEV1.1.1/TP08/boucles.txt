#include<stdlib.h>


int main(void) {
  int rep, i, j, k, l;
  printf("entre un nombre: \n");
  scanf("%d", &rep);
  for(i = 0; i <= rep; i++) {
    for(j = 0; j <= rep - i; j++ ) {
      for(k = 0; k <= rep - j; k++) {
        for(l = 0; l <= rep - k; l++) {
          if (i + j + k + l == rep) {
             printf("%d = %d + %d + %d + %d \n", rep, i, j, k, l);
          }
        }
      }
    }
  }
  return EXIT_SUCCESS;
}
