#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#define C 4
#define TAILLE 10

int main(void) {
    signed int tab[TAILLE], i, temp;
    srand(time(NULL));
    printf("+-----+-----+-----+-----+-----+-----+-----+-----+-----+-----+\n");
    for(i = 0; i < TAILLE; i++) {
        tab[i] = (rand() % 99) - 51;
        printf("| %3d ", tab[i]);
    }
    printf("|\n+-----+-----+-----+-----+-----+-----+-----+-----+-----+-----+\n");
    for (i = 0; i < TAILLE - C; i++) {
        temp = tab[i];
        tab[i] = tab[i + C];
        tab[i + C] = temp;
    }
    printf("+-----+-----+-----+-----+-----+-----+-----+-----+-----+-----+\n");
    for(i = 0; i < TAILLE; i++) {
        printf("| %3d ", tab[i]);
    }
    printf("|\n+-----+-----+-----+-----+-----+-----+-----+-----+-----+-----+\n");
    return 0;
}
