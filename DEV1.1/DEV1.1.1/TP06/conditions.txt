int main(void) {
    int annee;

    scanf("%d", &annee);
    if (annee % 400 == 0) {
        printf("%d est une année bissextile \n.", annee);
        return EXIT_SUCCESS;
    } else if (annee % 4 == 0) {
        if (annee % 100 != 0) {
            printf("%d est une année bissextile. \n", annee);
            return EXIT_SUCCESS;
        }
    }
    printf("%d n'est pas une année bissextile. \n", annee);
    return EXIT_SUCCESS;
}

version avec logique:
#include<stdio.h>
#include<stdlib.h>

int main(void) {
    int annee;
    printf("entrer une année \n");
    scanf("%d", &annee);
    ((annee % 4 == 0 && annee % 100 != 0) || annee % 400 == 0) ? printf("%d est une année bissextile. \n", annee) : printf("%d n'est pas une année bissextile. \n", annee);
    return EXIT_SUCCESS;
}
