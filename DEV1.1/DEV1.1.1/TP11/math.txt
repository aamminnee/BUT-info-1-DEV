#include <stdio.h>
#include <stdlib.h>
#include <math.h>

int main(void) {
  double res1 = sqrt(fabs(log(0.5)));
  double res2 = sin((M_PI)/6);
  double res3 = atan(pow(13, 2));
  double res4 = pow(exp(-1), 4);
  double res5 = log(-3);
  double res6 = pow(sqrt(2), 2);
  printf("res1 = %lg \n", res1);
  printf("res2 = %lg \n", res2);
  printf("res3 = %lg \n", res3);
  printf("res4 = %lg \n", res4);
  printf("res5 = %lg \n", res5);
  printf("res6 = %lg \n", res6);
  return EXIT_SUCCESS;
}

#include <stdio.h>
#include <stdlib.h>
#include <math.h>

int main(void) {
  double x1, x2, y1, y2, dist;
  printf("point 1 X \n");
  scanf("%lg", &x1);
  printf("point 1 Y \n");
  scanf("%lg", &y1);
  printf("point 2 X \n");
  scanf("%lg", &x2);
  printf("point 2 Y \n");
  scanf("%lg", &y2);
  dist = sqrt(pow((x2 - x1), 2) + pow((y2 - y1), 2));
  printf("la distance = %lg \n", dist);
  return EXIT_SUCCESS;
}

#include <stdio.h>
#include <stdlib.h>
#include <math.h>

int main(void) {
  double rep;
  int unite, dixieme;
  printf("entrer un reel: \n");
  scanf("%lf", &rep);
  unite = (int)trunc(rep) % 10;
  dixieme = (int)fabs(rep * 10) % 10;
  printf("unité: %d \n", unite);
  printf("dixième: %d \n", dixieme);
  return EXIT_SUCCESS;
}
