#include<stdio.h>
#include<stdlib.h>


int main(void) {
  int rep;
  int i, j;
  int elt = 1;
  int esp;
  printf("hauteur ? \n");
  scanf("%d", &rep);
  esp = rep - 1;
  for(i = 1; i <= rep; i++) {
    for(j = 1 ; j <= esp; j++) {
      printf(" ");
    }
    for(j = 1; j <= elt; j++) {
      printf("*");
    }
    elt += 2;
    esp--;
    printf("\n");
  }
  return EXIT_SUCCESS;
}

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <ctype.h>

int main(void) {
    int saut_de_lignes = 0, mots = 0, caracteres = 0;
    char rep[1000];

    printf("Écrire dans le fichier (Ctrl+D pour terminer):\n");

    while (scanf("%s", rep) != EOF) {
        caracteres += strlen(rep);
        mots++;
        if (strchr(rep, '\n')) {
            saut_de_lignes++;
        }
    }

    printf("Saut de lignes: %d\n", saut_de_lignes);
    printf("Mots: %d\n", mots);
    printf("Caractères: %d\n", caracteres);

    return EXIT_SUCCESS;
}
